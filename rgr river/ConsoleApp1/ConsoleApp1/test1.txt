#include <stdio.h>
#include <stdlib.h>
#include <Math.h>

/*С начала суток прошло N секунд (N – целое). 
Вывести текущее время 
			в формате час:минута:секунда.*/

int main() 
{
	int N, Chas, Minute;//где N=кол-во секунд
	printf("Введите кол-во секнуд пройденных с начала суток: ");
	scanf_s("%d",&N);
	Chas = N / 3600;//Где 3600 кол-во сек в часу
	Minute = (N / 60) % 60;//Без остаточного деления,вывод будет больше кол-ва минут в часе 
	N = N % 60;//Находим остаточные числа,для записи в сек

	printf("Время: %d:%d:%d ",Chas,Minute,N);



}
